Problem Statement: https://www.geeksforgeeks.org/count-number-binary-strings-without-consecutive-1s/

```
#include <bits/stdc++.h>
using namespace std;

int find(int n)
{
    vector<int> dp0(n+1,0);
    vector<int> dp1(n+1,0);
    dp0[1]=dp1[1]=1;
    for(int i=2;i<=n;i++)
    {
        dp0[i]=dp0[i-1]+dp1[i-1];
        dp1[i]=dp0[i-1];
    }
    return dp0[n]+dp1[n];
}

int main()
{
    int n;
    cin>>n;
    cout<<find(n);
}
```
