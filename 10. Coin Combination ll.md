Problem Statement: https://cses.fi/problemset/task/1636

```
#include <bits/stdc++.h>
using namespace std;

int find(vector<int>& coin,int n)
{
    vector<int> dp(n+1,0);
    dp[0]=1;
    for(auto x: coin)
    {
        for(int i=1;i<=n;i++)
        {
            if(i-x>=0){
                dp[i] += dp[i-x];
            }
        }
    }
    return dp[n];
}

int main()
{
    vector<int> coin ={2,3,5};
    int amount = 9;
    cout<<find(coin,amount);
}
```
